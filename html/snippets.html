Quicksort in Haskell...

quicksort' :: (Ord a) => [a] -> [a]
quicksort' [] = []
quicksort' (x:xs) =
    let bigger = quicksort' [a | a <- xs, a > x]
        smaller = quicksort' [a | a <- xs, a <= x]
    in  bigger ++ [x] ++ smaller
