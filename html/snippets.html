Quicksort in Haskell...

quicksort' :: (Ord a) => [a] -> [a]
quicksort' [] = []
quicksort' (x:xs) =
    let bigger = quicksort' (filter (>x) xs)
        smaller = quicksort' (filter (<=x) xs)
    in  bigger ++ [x] ++ smaller
